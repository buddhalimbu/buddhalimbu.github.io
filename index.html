UseFul Blogger

You can check the most useful blogger code
that might be helpful you to understand the basic blogger template design. 

                            

Link Condition

<--Print only in homepage which will create <a tag with href-->

<b:tag expr:href='data:blog.homepageUrl' name='a'>
    <data:blog.title/>
</b:tag>


Title link in diffrent page

<--if it is homepage create span tag-->

<b:tag cond='data:view.isHomepage' name='span'>

<--If it is post then create anchor tag-->

<b:tag cond='data:view.isPost or data:view.isSingleItem' name='a' expr:href='data:blog.homepageUrl'>
<data:blog.title/>
</b:tag>
</b:tag>
                            
                        
                            

Adding Class

<div class="header">

<--It will create header-homepage class to the header in homepage view-->
<b:class cond='data:view.isHomepage' name='header-homepage'/> 

<--It will add item-header class to the header in item or post view-->
<b:class cond='data:view.isPost' name='item-header-'/>

</div>
                            
                        
                            

Hide title in certain section

<b:includable id='widget-title'>
<b:if cond='data:defaultTitle or data:title'>

<--If section Id is header menu then title will no fetch anymore-->

<b:if cond='data:widget.sectionId not in ["header-menu"]'>
</b:if>
</b:if>
</b:includable>
                            
                        
                            

Get single post label

<--b if post have label and get first label only-->

<b:if cond='data:post.labels'>
<a class='post-label' expr:href='data:label.url' rel='tag'><data:post.labels.first.name/>
</a>

<--If has no any labe in post then print no label-->

<b:else/>
<span class='no-label' style='color:#ddd'> No label</span>
</b:if>        
                            
                        
                            

Custom link-style in per section

<--If section id is header then create ul based on your demand-->

<b:if cond='data:widget.sectionId == "header"'>

<nav id='header-main-menu-nav' itemscope='itemscope' itemtype='http://schema.org/SiteNavigationElement' role='navigation'>

<ul id='main-nav' role='menubar'>
<b:loop values='data:links' var='link'>
<li><a expr:data-text='data:link.name' expr:href='data:link.target' role='menuitem'><data:link.name/></a></li>
</b:loop>
</ul>
</nav>

<--Id section Id is header2 then add class and create different ul and nav -->

<b:elseif cond='data:widget.sectionId == "header2"'/>

<nav id='header-menu-nav' itemscope='itemscope' itemtype='http://schema.org/SiteNavigationElement' role='navigation'>

<ul class='topnav-menu'>
<b:loop values='data:links' var='link'>
<li><a expr:href='data:link.target'><data:link.name/></a></li>
</b:loop>
</ul>
</nav>

<--If not print this default value-->

<b:else/>
<nav>
<ul>
<b:loop values='data:links' var='link'>
<li><a expr:data-text='data:link.name' expr:href='data:link.target' role='menuitem'>
<data:link.name/>
</a>
</li>
</b:loop>
</ul>
</nav>
</b:if>
